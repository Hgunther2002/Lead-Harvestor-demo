    <button onclick="showLeads()">Search</button>


    <div id="results"></div>


    <script>

        function showLeads() {

            const zip = document.getElementById("zipcode").value;

            const resultsDiv = document.getElementById("results");


            const demoLeads = [

                { source: "circuit7.net", type: "probate-court", title: "Clay County Probate Division", url: "https://www.circuit7.net/probate/", address: "Clay County, MO", notes: "Official county probate page" },

                { source: "courts.mo.gov", type: "divorce-docket", title: "Clay County Divorce Dockets", url: "https://www.courts.mo.gov/cnet/welcome.do", address: "Clay County, MO", notes: "Statewide public docket portal" }

            ];


            if (zip === "64118") {

                let table = "<table><tr><th>Source</th><th>Type</th><th>Title</th><th>Address</th><th>Notes</th></tr>";

                demoLeads.forEach(lead => {

                    table += `<tr>

                        <td>${lead.source}</td>

                        <td>${lead.type}</td>

                        <td><a href="${lead.url}" target="_blank">${lead.title}</a></td>

                        <td>${lead.address}</td>

                        <td>${lead.notes}</td>

                    </tr>`;

                });

                table += "</table>";

                resultsDiv.innerHTML = table;

            } else {

                resultsDiv.innerHTML = "<p>No leads found for this ZIP code.</p>";

            }

        }

    </script>
